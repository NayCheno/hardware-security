\begin{frame}[t,shrink=6]{Arm Confidential Computing Architecture：研究背景}
\DeckKicker{研究背景}
\SlideMeta{li2022cca}{E1 / Foundational}{local\_pdf\_verified}
{\large\bfseries\color{Ink} CCA 要解决的矛盾是云平台仍要管理硬件资源，但 tenant 不再愿意信任 hypervisor 和 host OS。\par}\vspace{1.8mm}
\begin{columns}[T,totalwidth=\textwidth]
  \begin{column}{0.45\textwidth}
    \fcolorbox{Rule}{Paper}{%
  \begin{minipage}[t]{0.97\linewidth}
    \vspace{1.1mm}
    \hspace{1.4mm}\begin{minipage}{0.92\linewidth}
      \PanelTitle{讲解逻辑}
      \scriptsize \begin{itemize}
  \item 敏感数据越来越多地运行在 commodity cloud provider 上。
  \item 传统虚拟化把 hypervisor 作为强 TCB，但 hypervisor/OS 代码量大且漏洞多。
  \item 完全移除 host 管理能力又不现实，因为云平台仍需调度 CPU、内存和设备。
  \item CCA 的设计点是在资源管理与安全隔离之间划边界。
\end{itemize}
    \end{minipage}
    \vspace{1mm}
  \end{minipage}}
  \end{column}

      \scriptsize {\tiny
\begin{tabularx}{\linewidth}{@{}YYY@{}}
\toprule
\textbf{对象} & \textbf{传统 VM} & \textbf{CCA Realm} \\
\midrule
host OS/hypervisor & 可信管理者 & 不可信资源管理者 \\
tenant memory & host 可管理且可观察 & host 管理但不能直接访问 \\
firmware/monitor & 平台启动辅助 & RMM 成为隔离 enforcement TCB \\
安全目标 & 虚拟化隔离 & 机密性和完整性隔离 \\

\begin{frame}[t,shrink=6]{Arm Confidential Computing Architecture：解决方案}
\DeckKicker{解决方案}
\SlideMeta{li2022cca}{E1 / Foundational}{local\_pdf\_verified}
{\large\bfseries\color{Ink} CCA 的机制关键是 granule ownership 与 RMM mediation，把 host 操作变成受检查的 lifecycle 请求。\par}\vspace{1.8mm}
\begin{columns}[T,totalwidth=\textwidth]
  \begin{column}{0.45\textwidth}
    \fcolorbox{Rule}{Paper}{%
  \begin{minipage}[t]{0.97\linewidth}
    \vspace{1.1mm}
    \hspace{1.4mm}\begin{minipage}{0.92\linewidth}
      \PanelTitle{讲解逻辑}
      \scriptsize \begin{itemize}
  \item 内存以 granule 为基本管理单位，不同 physical address space 表示不同安全状态。
  \item host 不能直接改变 Realm 受保护状态，而要通过 RMI 调 RMM。
  \item Realm 内软件通过 RSI 与 RMM 交互，避免把 host 放回 trusted path。
  \item 论文还强调 verification，说明 CCA 设计不仅是接口集合，也需要可验证的访问控制不变量。
\end{itemize}
    \end{minipage}
    \vspace{1mm}
  \end{minipage}}
  \end{column}
  \begin{column}{0.49\textwidth}
    \fcolorbox{Rule}{Panel}{%
  \begin{minipage}[t]{0.97\linewidth}
    \vspace{1.1mm}
    \hspace{1.4mm}\begin{minipage}{0.92\linewidth}
      \PanelTitle{两条接口路径}
      \scriptsize \textbf{host path}\par
host/hypervisor → RMI call → RMM check → granule lifecycle update\par
{\tiny\textcolor{Muted}{资源管理被 RMM mediation}}\par\vspace{1.8mm}
\textbf{realm path}\par
Realm payload → RSI call → RMM service → attestation/lifecycle response\par
{\tiny\textcolor{Muted}{tenant 不直接信任 host}}\par\vspace{1.8mm}
    \end{minipage}
    \vspace{1mm}
  \end{minipage}}
  \end{column}
\end{columns}
\vspace{1mm}
{\tiny\textcolor{Muted}{Source: 接口路径依据论文对 RMI/RSI 和 granule access policy 的描述重绘。}}
\end{frame}
