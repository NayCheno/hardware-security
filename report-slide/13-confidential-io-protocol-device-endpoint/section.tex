\begin{frame}[t,shrink=6]{SPDM device identity and measurement：解决方案}
\DeckKicker{解决方案}
\SlideMeta{dmtf\_spdm\_2025}{E0 / Spec-standard SOTA}{local\_pdf\_verified}
{\large\bfseries\color{Ink} SPDM 把设备信任建立成 transcript-bound protocol，而不是一次性读取设备 ID；无新实验。\par}\vspace{1.8mm}
\begin{columns}[T,totalwidth=\textwidth]

    \fcolorbox{Rule}{Paper}{%

      \PanelTitle{讲解逻辑}
      \scriptsize \begin{itemize}
  \item GET\_CERTIFICATE 和 digest 消息提供证书链和 slot 状态。
  \item CHALLENGE\_AUTH 把随机 challenge、证书和签名绑定到 transcript。
  \item GET\_MEASUREMENTS 返回 measurement blocks 和 summary hash。
  \item KEY\_EXCHANGE/FINISH 建立后续 secured control messages。
\end{itemize}
    \end{minipage}
    \vspace{1mm}
  \end{minipage}}

      \scriptsize \begin{itemize}
  \item 01. negotiate version/capabilities
  \item 02. retrieve certificate chain
  \item 03. challenge responder
  \item 04. request measurements
  \item 05. derive session keys
  \item 06. protect later messages

\begin{frame}[t,shrink=6]{Separate but Together: TLS channel binding with remote attestation：解决方案}
\DeckKicker{解决方案}
\SlideMeta{weinhold2025tlsra}{E1 / Peer-reviewed endpoint channel-binding evidence}{local\_pdf\_verified}
{\large\bfseries\color{Ink} TLS+RA 的设计是 double linking：TLS private key 绑定 transcript，attestation key 绑定同一 handshake context。\par}\vspace{1.8mm}
\begin{columns}[T,totalwidth=\textwidth]
  \begin{column}{0.45\textwidth}
    \fcolorbox{Rule}{Paper}{%
  \begin{minipage}[t]{0.97\linewidth}
    \vspace{1.1mm}
    \hspace{1.4mm}\begin{minipage}{0.92\linewidth}
      \PanelTitle{讲解逻辑}
      \scriptsize \begin{itemize}
  \item TLS side 保持标准 TLS 安全属性和 failure behavior。
  \item RA side 使用 nonce 和 report 把 TEE state 放入 handshake context。
  \item 两条签名链使用不同 key，但来自同一 message log。
  \item 这样可以做到 TEE-agnostic，并支持 client-side 或 mutual attestation 扩展。
\end{itemize}
    \end{minipage}
    \vspace{1mm}
  \end{minipage}}
  \end{column}
  \begin{column}{0.49\textwidth}
    \fcolorbox{Rule}{Panel}{%
  \begin{minipage}[t]{0.97\linewidth}
    \vspace{1.1mm}
    \hspace{1.4mm}\begin{minipage}{0.92\linewidth}
      \PanelTitle{double linking}
      \scriptsize \begin{itemize}
  \item 01. TLS transcript accumulates messages
  \item 02. TLS key signs transcript
  \item 03. attestation nonce enters report
  \item 04. TEE report signs measurement/context
  \item 05. verifier checks symmetry
\end{itemize}
    \end{minipage}
