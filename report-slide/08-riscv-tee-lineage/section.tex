\begin{frame}[t,shrink=6]{SPEAR-V: low-overhead RISC-V enclave primitive：解决方案}
\DeckKicker{解决方案}
\SlideMeta{schrammel2023spearv}{E1 / Peer-reviewed SOTA}{local\_pdf\_verified}
{\large\bfseries\color{Ink} SPEAR-V 把 enclave ID/tag 放进 memory translation metadata，让不可信 OS 仍管理 mapping 但不能越权访问。\par}\vspace{1.8mm}
\begin{columns}[T,totalwidth=\textwidth]
  \begin{column}{0.45\textwidth}
    \fcolorbox{Rule}{Paper}{%
  \begin{minipage}[t]{0.97\linewidth}
    \vspace{1.1mm}
    \hspace{1.4mm}\begin{minipage}{0.92\linewidth}
      \PanelTitle{讲解逻辑}
      \scriptsize \begin{itemize}
  \item SM 给 enclave page 打 tag，并保留特殊 enclave ID 给 SM 自身。
  \item TLB entry 存储 memory tag fields。
  \item page table walk 时检查 tag，OS 管 mapping 但访问受硬件约束。
  \item 这种设计把资源管理和访问 enforcement 分开。
\end{itemize}
    \end{minipage}
    \vspace{1mm}
  \end{minipage}}
  \end{column}
  \begin{column}{0.49\textwidth}
    \fcolorbox{Rule}{Panel}{%
  \begin{minipage}[t]{0.97\linewidth}
    \vspace{1.1mm}
    \hspace{1.4mm}\begin{minipage}{0.92\linewidth}
      \PanelTitle{tag-aware translation}
      \scriptsize \begin{itemize}
  \item 01. SM creates enclave context
  \item 02. tag pages with enclave ID
  \item 03. OS maintains page tables
  \item 04. TLB stores tag metadata
  \item 05. hardware rejects wrong tag
\end{itemize}
    \end{minipage}
